-- TECNOMULIPAST · Schema PostgreSQL/Supabase
CREATE TABLE odp (
  id SERIAL PRIMARY KEY,
  programma TEXT NOT NULL,
  commessa TEXT NOT NULL,
  des_prod TEXT,
  articolo TEXT,
  quant_da_produrre INTEGER NOT NULL,
  stato TEXT CHECK (stato IN ('in_attesa','in_lavorazione','completato')),
  sync_robot BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE avanzamento (
  id BIGSERIAL PRIMARY KEY,
  odp_id INTEGER REFERENCES odp(id),
  data_gg INTEGER, data_mm INTEGER, data_aa INTEGER,
  hh INTEGER, min INTEGER, sec INTEGER,
  timestamp_calc TIMESTAMPTZ GENERATED ALWAYS AS (
    MAKE_TIMESTAMP(data_aa+2000, data_mm, data_gg, hh, min, sec)
  ) STORED,
  operatore TEXT, stazione INTEGER,
  programma TEXT, commessa TEXT, articolo TEXT,
  tempo_esecuzione REAL,
  qta_prodotta INTEGER,
  flag_finito INTEGER,          -- 1 | 199 | 299
  id_sql INTEGER,               -- FK → odp.id
  fbk_volt REAL, fbk_amp REAL,
  cmd_wfs REAL,
  arc_on_h INTEGER, arc_on_m INTEGER, arc_on_s INTEGER,
  allarme1 TEXT, allarme2 TEXT, allarme3 TEXT,
  allarme4 TEXT, allarme5 TEXT, allarme6 TEXT,
  allarme7 TEXT, allarme8 TEXT, allarme9 TEXT, allarme10 TEXT,
  is_diagnostica BOOLEAN GENERATED ALWAYS AS (allarme1 IS NOT NULL) STORED
);

CREATE TABLE kpi_qualita (
  id BIGSERIAL PRIMARY KEY,
  periodo_da TIMESTAMPTZ, periodo_a TIMESTAMPTZ,
  programma TEXT, articolo TEXT, n_cicli INTEGER,
  volt_medio REAL, amp_medio REAL, wfs_medio REAL,
  arc_on_sec_totale INTEGER,
  tasso_nc REAL, oee_saldatura REAL, n_allarmi INTEGER
);

CREATE INDEX idx_avanzamento_odp ON avanzamento(odp_id);
CREATE INDEX idx_avanzamento_ts ON avanzamento(timestamp_calc);
CREATE INDEX idx_avanzamento_flag ON avanzamento(flag_finito);
